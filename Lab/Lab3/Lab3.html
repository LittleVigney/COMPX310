# Preprocessing: OneHot for categorical, StandardScaler for numerical (Use ColumnTransformer)
preprocess = ColumnTransformer(transformers=[
    ('num', StandardScaler(), numeric_cols),
    ('cat', OneHotEncoder(drop="first", handle_unknown="ignore"), categorical_cols)
])

# Final pipeline with Logistic Regression (Use Pipeline)
pipeline_model = Pipeline([
    ('preprocessor_model', preprocess),
    ('classify_model', LogisticRegression(random_state=6327))
])

# Try different values of C (regularization strength)
C_values = [0.01, 0.1, 1, 10, 100]
results = []

for C_value in C_values:
    pipeline_model = Pipeline([
        ("preprocessor_model", preprocess),
        ("classify_model", LogisticRegression(C=C_value, random_state=6327))
    ])
    
    pipeline_model.fit(X_training, y_training)
    
    accuracy = pipeline_model.score(X_test, y_test)
    results.append({
        "C value": C_value,
        "accuracy": accuracy
    })

# Display results
print("C value     accuracy")
for result in results:
    print(f"{result["C value"]}     {result["accuracy"]}")

Why do we save the model as .joblib and load it later in app.py instead of retraining the model every time?

Retraining the model will cost more time to get the prediction result. Saving the model could save time and computation resources.
